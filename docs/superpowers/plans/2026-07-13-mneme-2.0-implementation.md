# Mneme 2.0 — LLM Wiki Refactor Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.
>
> **Goal:** Restore Mneme to the LLM Wiki thesis — OKF + tagged Markdown as the only source of truth, with `dream`/`search` as the only user intents — and ship 2.0.0 as a single skill zip with disposable SQLite FTS / sqlite-vec / BGE accelerators.
>
> **Architecture:** Dream (write) and Search (read) become the user-facing intents; Python CLI carries deterministic operations only (`init / lint / reindex / search --bundle --json`). Optional L2 (sqlite-vec + FastEmbed + BGE) is fetched on first `reindex --l2` via a documented bootstrap step. Dream defaults to dry-run; `--apply` is the only path to writes. Indexes are disposable — Markdown remains the source of truth.
>
> **Tech Stack:** Python 3.11+ stdlib (sqlite3, tomllib, zipfile, configparser), `PyYAML` (optional fallback), `sqlite-vec` + `fastembed` (opt-in L2 only), `BAAI/bge-small-zh-v1.5` (default embedder), standard `Makefile`-driven pytest suite.

## Global Constraints

- Target version: **`2.0.0`**. `__version__` in `skills/mneme/scripts/mneme/__init__.py`, frontmatter `version` in `skills/mneme/SKILL.md`, and zip file name must match exactly `mneme-2.0.0.zip`. `tests/test_version.py` enforces this.
- OKF v0.1 compliance remains the only must-confirm rule on bundle format (three MUSTs from `.research/upstream/OKF-SPEC.md:343-349`). Mneme does **not** add MUSTs to OKF; `tags` is a Mneme writer rule, not a protocol rule.
- Delivery model is **single skill zip** at `dist/mneme-2.0.0.zip`, unpacked to `~/.claude/skills/mneme/`. No wheel, no console command, no `mneme` import-by-name. `tests/test_release_layout.py` enforces this.
- L0 (index/tags/links/grep) and L1 (sqlite3 + FTS5, default on first `reindex`) are zero-dep. L2 (`sqlite-vec` + FastEmbed) is opt-in. No auto `pip install`. No auto-downloaded ONNX bundled into the zip.
- User-facing verbs in SKILL/README/introduction: **`dream` and `search` only**. CLI `init/lint/reindex/search` are internal/agent-only.
- `dream` default = preview/dry-run. Writes require `--apply`. Subprocess must never call `git add -A`, push, or branch; non-git repos must not error.
- Error philosophy: keep OKF v0.1 §9 tolerance consumer contract — one bad file never invalidates the bundle. External OKF missing `tags` is a WARN, not a failure.
- All Python source is in `skills/mneme/scripts/mneme/`. Skill entry file is `scripts/mneme.py`; call sites may also use `python3 -m mneme`. Tests live in `tests/`. PyYAML is optional; without it, lint still runs (warn-only strict mode).
- `pyproject.toml` MUST stay minimal: only `[tool.pytest.ini_options]`. Extras (`index`, `validate`, `toml10`, `converters`) are loaded at runtime via `try/except ImportError`.
- CI must not run `python -m build --wheel`, `pip install -e '.[...]'`, or `pip install dist/mneme-*.whl`. Only `pytest -m 'not network and not compat'` plus zip integrity checks.
- Every task ends in an independently verifiable state — green tests for the unit touched, no orphan files, no broken links in introduction.

class L2UnavailableError(RuntimeError):
    pass


def ensure_schema(conn):
    conn.executescript(
        """
        CREATE TABLE IF NOT EXISTS pages (
            id INTEGER PRIMARY KEY,
            path TEXT UNIQUE NOT NULL,
            type TEXT,
            title TEXT,
            description TEXT,
            tags TEXT,
            mtime REAL
        );
        CREATE VIRTUAL TABLE IF NOT EXISTS pages_fts USING fts5(
            title, description, tags, body, content='pages', content_rowid='id'
        );
        CREATE TRIGGER IF NOT EXISTS pages_ai AFTER INSERT ON pages BEGIN
            INSERT INTO pages_fts(rowid, title, description, tags, body)
            VALUES (new.id, new.title, new.description, new.tags, '');
        END;
        CREATE TRIGGER IF NOT EXISTS pages_ad AFTER DELETE ON pages BEGIN
            INSERT INTO pages_fts(pages_fts, rowid, title, description, tags, body)
            VALUES('delete', old.id, old.title, old.description, old.tags, '');
        END;
        CREATE TRIGGER IF NOT EXISTS pages_au AFTER UPDATE ON pages BEGIN
            INSERT INTO pages_fts(pages_fts, rowid, title, description, tags, body)
            VALUES('delete', old.id, old.title, old.description, old.tags, '');
            INSERT INTO pages_fts(rowid, title, description, tags, body)
            VALUES (new.id, new.title, new.description, new.tags, '');
        END;
        """
    )


def create_fts_index(conn):
    # No-op when empty; FTS5 stays consistent via triggers.
    conn.execute("SELECT 0 FROM pages_fts LIMIT 0")


def embedder_for_l2(model: str = "BAAI/bge-small-zh-v1.5"):
    try:
        from fastembed import TextEmbedding  # type: ignore
    except Exception as exc:
        raise L2UnavailableError(
            "L2 (sqlite-vec + FastEmbed) is not installed. "
            "Run: pip install 'sqlite-vec>=0.1.9,<0.2' 'fastembed>=0.8,<0.9'"
        ) from exc

    def _embed(texts):
        emb = TextEmbedding(model_name=model)
        return [e.tolist() for e in emb.embed(texts)]

    return _embed
```

def reindex_paths(paths, bundle: Path, l2: bool = False):
    import sqlite3
    from .okflib import parse_frontmatter
    db_dir = Path(bundle) / ".mneme"
    db_dir.mkdir(exist_ok=True)
    db = db_dir / "index.db"
    conn = sqlite3.connect(db)
    ensure_schema(conn)
    for p in paths:
        rel = str(p.relative_to(bundle))
        text = p.read_text()
        fm, _ = parse_frontmatter(text)
        try:
            body_start = text.index("---", 3) + 3
        except ValueError:
            body_start = 0
        body = text[body_start:]
        tags = ",".join(fm.get("tags") or [])
        conn.execute(
            "INSERT OR REPLACE INTO pages (path, type, title, description, tags) VALUES (?,?,?,?,?)",
            (rel, fm.get("type"), fm.get("title"), fm.get("description"), tags),
        )
    conn.commit()
    if l2:
        embedder = embedder_for_l2()
        # L2 indexing wired through sqlite-vec; reuse existing indexlib.search code
        _build_vec_index(conn, [str(p.relative_to(bundle)) for p in paths], embedder)

def lint_bundle(bundle: Path, *, require_tags: bool = True, okf_external: bool = False):
    diagnostics = []
    # OKF MUSTs: every non-reserved .md must have frontmatter with non-empty type
    for p in Path(bundle).rglob("*.md"):
        rel = p.relative_to(bundle)
        if rel.name in {"index.md", "log.md"}:
            # Reserved file semantics
            if rel.parent == Path("."):
                if rel.name == "index.md":
                    body = p.read_text()
                    fm, _ = parse_frontmatter(body)
                    if fm and "okf_version" in fm and fm["okf_version"] != "0.1":
                        diagnostics.append({"severity": "ERROR", "code": "OKF-INDEX-VERSION", "path": str(rel)})
            if rel.name == "log.md":
                # All headings must be newest-first ISO 8601
                for line_no, line in enumerate(p.read_text().splitlines(), 1):
                    if line.startswith("#") and not _IS_DATE.search(line):
                        diagnostics.append({"severity": "ERROR", "code": "OKF-LOG-FORMAT", "path": f"{rel}:{line_no}"})
            continue
        text = p.read_text()
        if not text.startswith("---\n"):
            diagnostics.append({"severity": "ERROR", "code": "OKF-MUST-1-FRONTMATTER", "path": str(rel)})
            continue
        fm, body = parse_frontmatter(text)
        if not fm or not (fm.get("type") or "").strip():
            diagnostics.append({"severity": "ERROR", "code": "OKF-MUST-2-TYPE", "path": str(rel)})
        elif require_tags and not okf_external and not fm.get("tags"):
            diagnostics.append({"severity": "ERROR", "code": "MNEME-TAG-MISSING", "path": str(rel)})
        elif okf_external and not fm.get("tags"):
            diagnostics.append({"severity": "WARN", "code": "MNEME-TAG-MISSING", "path": str(rel)})
    return {"diagnostics": diagnostics}


import re
_IS_DATE = re.compile(r"^##\s+\d{4}-\d{2}-\d{2}\b")
```

```python
#!/usr/bin/env python3
"""Build dist/mneme-<version>.zip with deterministic exclusions."""
from __future__ import annotations

import argparse
import re
import zipfile
from pathlib import Path

ROOT = Path(__file__).resolve().parents[1]
SRC = ROOT / "skills" / "mneme"
PKG_INIT = SRC / "scripts" / "mneme" / "__init__.py"

EXCLUDE_DIR = {"__pycache__", ".egg-info", ".mneme", ".git"}
EXCLUDE_EXT = {".pyc", ".pyo", ".DS_Store"}


def _version() -> str:
    text = PKG_INIT.read_text()
    m = re.search(r'^__version__\s*=\s*"([^"]+)"', text, re.M)
    if not m:
        raise SystemExit(f"version not found in {PKG_INIT}")
    return m.group(1)


def main() -> int:
    ap = argparse.ArgumentParser()
    ap.add_argument("--out", default=str(ROOT / "dist"))
    args = ap.parse_args()
    out_dir = Path(args.out)
    out_dir.mkdir(parents=True, exist_ok=True)
    for old in out_dir.glob("mneme-*.zip"):
        old.unlink()
    version = _version()
    out_zip = out_dir / f"mneme-{version}.zip"
    with zipfile.ZipFile(out_zip, "w", zipfile.ZIP_DEFLATED, compresslevel=9) as zf:
        for p in sorted(SRC.rglob("*")):
            if not p.is_file():
                continue
            rel = p.relative_to(SRC)
            parts = set(rel.parts)
            if parts & EXCLUDE_DIR or p.suffix in EXCLUDE_EXT:
                continue
            zf.write(p, arcname=f"mneme/{rel.as_posix()}")
    print(out_zip)
    return 0


if __name__ == "__main__":
    raise SystemExit(main())
```
